Primero, revisemos la version extendida de este algoritmo
```pseudo
\begin{algorithm}
\caption{BFS-Aumentado}
\begin{algorithmic}
\State Marcar como explorado $s$
\State $l(s)=0, l(v)=\infty, v\neq s$
\State $Q:=$ cola inicializada con $s$
\While{Q no vacio}
	\State $w$ vertice del frente que extraeremos
	\For{cada $(v,w)$ en la lista de adyacencia de v} 
		\If{$w$ no explorado}
			\State marcar como explorado
			\State $l(w):=l(v)+1$
			\State añadir $w$ al final de $Q$
		\EndIf
	\EndFor
\EndWhile
\end{algorithmic}
\end{algorithm}
```
